## 1. QIIME2 environment
source activate qiime2-2022.8
#
# 2. Merge ASV tables
qiime feature-table merge \
    --i-tables ~/Documents/Stress_tolerance/amplicon_data/working_qzas/BOARD/ITS/demux_BOARD_ITS_BS_DADA_out/table.qza \
    --i-tables ~/Documents/Stress_tolerance/amplicon_data/working_qzas/BOARD/ITS/demux_BOARD_ITS_RE_DADA_out/table.qza \
    --i-tables ~/Documents/Stress_tolerance/amplicon_data/working_qzas/BOARD/ITS/demux_BOARD_ITS_RH_DADA_out/table.qza \
    --o-merged-table ~/Documents/Stress_tolerance/amplicon_data/final_qzas/BOARD/ITS/BOARD_ITS_merged_table.qza
#
# 3. Merge representative sequences
qiime feature-table merge-seqs \
    --i-data ~/Documents/Stress_tolerance/amplicon_data/working_qzas/BOARD/ITS/demux_BOARD_ITS_BS_DADA_out/representative_sequences.qza \
    --i-data ~/Documents/Stress_tolerance/amplicon_data/working_qzas/BOARD/ITS/demux_BOARD_ITS_RE_DADA_out/representative_sequences.qza \
    --i-data ~/Documents/Stress_tolerance/amplicon_data/working_qzas/BOARD/ITS/demux_BOARD_ITS_RH_DADA_out/representative_sequences.qza \
    --o-merged-data ~/Documents/Stress_tolerance/amplicon_data/final_qzas/BOARD/ITS/BOARD_ITS_merged_repseqs.qza
#
## 4. Deactivate QIIME2 environment
conda deactivate
